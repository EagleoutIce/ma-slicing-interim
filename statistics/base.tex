\def\DrawMinimum{0.000000001}

% overwrite default so that we get 0 for unknown
\pgfkeysdef{/pgfplots/table/@undefined column text}{0}%
\pgfplotsset{
   HistAxis/.style={
      ymin=0, height=10cm,width=5cm, ymax=1000,%  xmax=1000,
      xmin=-1, xmax=1, % plotted as density
      xtick={-1,-0.5,0,0.5,1},
      xticklabels={1,0.5,0,0.5,1},
      x tick label style={font=\sffamily\footnotesize},
      % axis on top,
      xbar interval,
      ymajorgrids=false, xmajorgrids=false,
      scaled ticks=false,
      ymajorticks=false,
      axis lines=middle,
      every inner x axis line/.append style={lgray,-},
      every inner y axis line/.append style={lgray,-}
   },
   % column as #1
   TableData/.style={
      table/col sep=comma,
      table/y={to},
      table/x expr={\thisrow{#1}},
      % do not draw too-close to 0-bars
      restrict expr to domain={\thisrow{#1}}{\DrawMinimum:1}
   },
   % version for the other half of the plot
   TableData'/.style={
      TableData=#1,
      table/x expr={-\thisrow{#1}},
      restrict expr to domain={\thisrow{#1}}{\DrawMinimum:1}
   },
   PlotDraw/.style={
      fill,#1,
   }
}

% #1 column index | #2 title
\def\PlotForColumn#1#2{%
\begin{tikzpicture}[line join=round,line cap=round]
\begin{axis}[HistAxis,title style={at={(0.5,0)},anchor=north,yshift=-6mm},title={#2}]
\addplot+[PlotDraw={lightgray},forget plot] table [TableData={#1}] {collected/cran-assignments.dat};
\addplot+[PlotDraw={paletteA}] table [TableData'={#1}] {collected/social-assignments.dat};
\end{axis}
\end{tikzpicture}%
}
